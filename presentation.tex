\section{Problem \& Data}

\begin{frame}{Problem Statement}
  \begin{block}{Task}
    Every trade has an \textbf{aggressor} — the side that crossed the spread.
    The public feed often omits this label. We reconstruct it.
  \end{block}

  \medskip
  \begin{columns}[T]
    \column{0.48\textwidth}
    \textbf{Label convention}
    \begin{itemize}
      \item \code{True}  = \textbf{sell aggressor} (hit the bid)
      \item \code{False} = \textbf{buy aggressor} (lifted the ask)
    \end{itemize}
    \medskip
    \textbf{Why it matters}
    \begin{itemize}
      \item VPIN, Kyle's $\lambda$, order-flow imbalance
      \item Effective spread decomposition
    \end{itemize}

    \column{0.48\textwidth}
    \begin{exampleblock}{Deliverable}
      \small
      \begin{center}
        \texttt{classify\_side(trades)}\\
        $\to$ \texttt{pd.Series[bool]}
      \end{center}
      \smallskip
      Input: price, amount, time only\\
      {\footnotesize (no order book at inference time)}
    \end{exampleblock}
  \end{columns}
\end{frame}

% ─────────────────────────────────────────────────────────────
\begin{frame}{Dataset}
  \begin{columns}[T]
    \column{0.52\textwidth}
    \begin{itemize}
      \item \textbf{Symbols:} WIFUSDT, ZAMAUSDT (Binance perps)
      \item \textbf{Data:} trades + L2 order book (10 levels each side)
    \end{itemize}

    \medskip
    \begin{center}
    \begin{tabular}{llr}
      \toprule
      Split & Date & Trades \\
      \midrule
      Train      & Apr 12 &   41{,}967 \\
      Validation & Apr 13 &   47{,}276 \\
      \textbf{Test} & \textbf{Apr 14} & \textbf{607{,}949} \\
      \bottomrule
    \end{tabular}
    \end{center}

    \column{0.44\textwidth}
    \begin{block}{Key constraints}
      \begin{itemize}
        \item Order book = \textbf{oracle for training only}
        \item Final function: trades only
        \item Label balance $\approx 52\%$ sell / $48\%$ buy
        \item Test touched only at the end
      \end{itemize}
    \end{block}
  \end{columns}
\end{frame}

\section{Decision Tree}

\begin{frame}{Feature Engineering}
  \begin{columns}[T]
    \column{0.50\textwidth}
    \textbf{Trade-only features} (24 total)
    \begin{itemize}\setlength\itemsep{1pt}
      \item Tick direction: lags 1–5, forward-filled
      \item Log-returns: horizons 1, 3, 5, 10 trades
      \item Inter-trade gaps: $\Delta t$, $\log(1{+}\Delta t)$
      \item Volume z-score: rolling 10/20/50
      \item Run-length \& direction of current streak
      \item Price position in recent 20/50-trade range
    \end{itemize}

    \column{0.46\textwidth}
    \textbf{Midpoint features} (3 extra, from order book)
    \begin{itemize}
      \item \code{mid\_delta}      $= p - \text{mid}$
      \item \code{mid\_abs\_delta} $= |p - \text{mid}|$
      \item \code{mid\_sign}       $= \mathrm{sign}(p - \text{mid})$
    \end{itemize}

    \medskip
    \begin{alertblock}{Note}
      Midpoint features used in notebook comparison only.\\
      Deployed \code{classify\_side()} uses trades only.
    \end{alertblock}
  \end{columns}
\end{frame}

\section{Team Models}

\begin{frame}{XGBoost Methodology}

\small

\begin{block}{Core idea}
Instead of relying only on the classical Tick Rule, we trained an XGBoost classifier to learn nonlinear trade-flow dynamics from engineered features.
\end{block}

\smallskip
\textbf{Feature groups used:}
\begin{itemize}\setlength\itemsep{1pt}
    \item Tick direction and multi-lag ticks
    \item Short-term log returns
    \item Inter-trade timing
    \item Rolling volume z-scores
    \item Run-length and streak persistence
    \item Price position within recent local ranges
    \item Round-number proximity effects
\end{itemize}

\smallskip
\textbf{Validation setup:}
\begin{itemize}\setlength\itemsep{1pt}
    \item Train: 2026-04-12 \quad Val: 2026-04-13 \quad Hold-out Test: 2026-04-14
\end{itemize}

\smallskip
All features computed with strict no-look-ahead logic to avoid information leakage.

\end{frame}
